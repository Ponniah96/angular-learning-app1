Angular Forms

Forms

Angular Forms are the build in forms which is available in angular library to build simple as well as complex forms.

    
In app.module.ts
import { ReactiveFormsModule } from '@angular/forms';
@NgModule({
imports: [
    ReactiveFormsModule
],
})

In introduction.component.ts
import { FormBuilder } from '@angular/forms';
export class IntroductionComponent {
basicForm=this.formBuilder.group({
    name:'',
    age:0,
    gender:'',
    role:''
    });  
constructor(
    private formBuilder:FormBuilder,
    ){}  
onSubmit(): void {
    console.log('Form values: ',this.basicForm.value);
    this.basicForm.reset();
    }
}

In introduction.component.html
<form [formGroup]="basicForm" (ngSubmit)="onSubmit()">
<div>
    <label for="name">Name</label>
    <input id="name" placeholder="Enter Name"  type="text" formControlName="name">
</div>
<button type="submit" class="button">Submit</button>
</form>
