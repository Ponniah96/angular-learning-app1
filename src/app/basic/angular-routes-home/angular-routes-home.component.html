Router

Angular router is used to navigate angular app from single page to multiple apps

Example:


In app-routing-module.ts 
import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';

import { NavigationComponentComponent } from './navigation-component/navigation-component.component';
import { Navigation2Component } from './navigation2/navigation2.component';

const routes: Routes = [
{path:'nav1', component:NavigationComponentComponent},
{path:'nav2', component:Navigation2Component}
];

@NgModule({
imports: [RouterModule.forRoot(routes)],
exports: [RouterModule]
})
export class AppRoutingModule { }

In introduction.component.html
<a class="link" [routerLink]="['/nav1']">
Navigation Component 1
</a>
<a class="link" [routerLink]="['/nav2']">
Navigation Component 2
</a>
